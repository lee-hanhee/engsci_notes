    \section{Sequences and Series}
    A sequence $\{a_n\}$ is:
        \begin{itemize}
            \item increasing iff $a_n < a_{n+1}$
            \item non-decreasing iff $a_k \le a_{n+1}$
            \item decreasing iff $a_n > a_{n+1}$
            \item non-increasing iff $a_n \ge a_{n+1}$
        \end{itemize}
